'' : id; render(); }); bindOnce(button, 'keydown', event => { if (event.key === 'ArrowDown' || event.key === 'Enter' || event.key === ' ') { event.preventDefault(); state.openDropdown = button.dataset.dropdownTrigger; const multi = dropdown.dataset.multi === 'true'; setDropdownValue(scope, field, value, multi); if (!multi) state.openDropdown = ''; render(); }); bindOnce(option, 'keydown', handleDropdownOptionKeydown); }); } function bindKanban() { const toggle = document.getElementById('toggleGroups'); if (toggle) { toggle.addEventListener('change', event => { state.showGroupsInKanban = event.target.checked; render(); }); } document.querySelectorAll('[data-column-drag]').forEach(handle => { handle.addEventListener('dragstart', event => { state.columnDragId = handle.dataset.columnDrag; event.dataTransfer.effectAllowed = 'move'; 'drop-indicator-left' : 'drop-indicator-right'); }); columnEl.addEventListener('dragleave', event => { if (!columnEl.contains(event.relatedTarget)) { columnEl.classList.remove('drop-indicator-left', 'drop-indicator-right'); state.draggedNodeId = card.dataset.cardId; state.draggingCardId = card.dataset.cardId; event.dataTransfer.effectAllowed = 'move'; event.dataTransfer.setData('text/plain', state.draggedNodeId); card.classList.add('dragging'); }); document.querySelectorAll('[data-drop-column]').forEach(area => { area.addEventListener('dragover', event => { if (state.columnDragId) return; event.preventDefault(); state.dragOverColumnId = area.dataset.dropColumn; target[field] = value; } function setDropdownValue(scope, field, value, multi) { const target = scope === 'filters' ? state.filters : draftForScope(scope); if (!target) return; if (multi) { const current = Array.isArray(target[field]) ? target[field].slice() : [];
