// Filtering, hierarchy traversal, and gantt scale calculations. function filteredNodes() { return state.nodes.filter(node => matchesFilters(node)); } function matchesFilters(node) { const query = state.filters.query.trim().toLowerCase(); if (query) { const text = [node.name, node.description].join(' ').toLowerCase(); if (!text.includes(query)) return false; } if (state.filters.parentId && node.id !== state.filters.parentId && !isDescendantOf(node.id, state.filters.parentId)) return false; if ((state.filters.statusIds || []).length && !state.filters.statusIds.includes(node.statusColumnId)) return false; if (state.filters.assigneeId && !node.assigneeIds.includes(state.filters.assigneeId)) return false; if (state.filters.priority && node.priority !== state.filters.priority) return false; if (state.filters.dateStart && state.filters.dateEnd) { if (!node.startDate || !node.endDate) return false; if (node.endDate < state.filters.dateStart || node.startDate > state.filters.dateEnd) return false; } return true; } function visibleTreeNodes() { const roots = state.nodes.filter(node => !node.parentId).sort(compareNodeForList); const matching = new Set(state.nodes.filter(matchesFilters).map(node => node.id)); const includeSet = new Set(matching); matching.forEach(id => ancestorChain(id).forEach(ancestorId => includeSet.add(ancestorId))); if (state.shellNodeId && state.nodesById.has(state.shellNodeId)) { includeSet.add(state.shellNodeId); ancestorChain(state.shellNodeId).forEach(ancestorId => includeSet.add(ancestorId)); } const rows = []; function visit(node) { if (!includeSet.has(node.id)) return; if (node.id !== state.rootId) rows.push(node); if (node.id !== state.rootId && state.collapsed.has(node.id)) return; childrenOf(node.id).sort(compareNodeForList).forEach(visit); } roots.forEach(visit); return rows; } function treeOrderedNodes() { const rows = []; function visit(node) { rows.push(node); childrenOf(node.id).sort(compareNodeOrder).forEach(visit); } state.nodes.filter(node => !node.parentId).sort(compareNodeOrder).forEach(visit); return rows; } function childrenOf(parentId) { return state.nodes.filter(node => node.parentId === parentId).sort(compareNodeOrder); } function collectDescendants(nodeId) { const result = []; const stack = childrenOf(nodeId).map(node => node.id); while (stack.length) { const id = stack.shift(); result.push(id); childrenOf(id).forEach(child => stack.push(child.id)); } return result; } function ancestorChain(nodeId) { const chain = []; let node = state.nodesById.get(nodeId); while (node && node.parentId) { chain.push(node.parentId); node = state.nodesById.get(node.parentId); } return chain; } function isDescendantOf(nodeId, ancestorId) { return ancestorChain(nodeId).includes(ancestorId); } function depthOf(nodeId) { return ancestorChain(nodeId).length; } function taskDepthOf(nodeId) { return Math.max(0, depthOf(nodeId) - 1); } function childTaskDepth(parentId) { return parentId === state.rootId ? 0 : taskDepthOf(parentId) + 1; } function lastVisibleDescendantId(parentId) { const visibleIds = new Set(visibleTreeNodes().map(node => node.id)); let lastId = parentId; function visit(nodeId) { if (state.collapsed.has(nodeId)) return; childrenOf(nodeId).sort(compareNodeForList).forEach(child => { if (!visibleIds.has(child.id)) return; lastId = child.id; visit(child.id); }); } visit(parentId); return lastId; } function toggleCollapsed(nodeId) { if (state.collapsed.has(nodeId)) state.collapsed.delete(nodeId); else state.collapsed.add(nodeId); render(); } function visibleAnchor(nodeId, visibleSet) { let node = state.nodesById.get(nodeId); while (node && !visibleSet.has(node.id) && node.parentId) node = state.nodesById.get(node.parentId); return node ? node.id : nodeId; } function ganttScale(rows) { const scheduled = rows.filter(node => node.displayStartDate && node.displayEndDate); const today = localTodayDay(); const scaleDays = scheduled .flatMap(node => [dateToDay(node.displayStartDate), dateToDay(node.displayEndDate)]) .concat(ganttTimelineMarkerDates()) .concat(today); const filterRange = dateFilterRange(); if (filterRange) scaleDays.push(filterRange.start, filterRange.end); let startDay = Math.min(...scaleDays); let endDay = Math.max(...scaleDays); const padding = state.ganttZoom === 'day' ? { past: 30, future: 120 } : state.ganttZoom === 'week' ? { past: 60, future: 240 } : { past: 90, future: 365 }; startDay -= padding.past; endDay += padding.future; const baseDayWidth = state.ganttZoom === 'day' ? 18 : state.ganttZoom === 'week' ? 10 : 4; const scaleFactor = nearestGanttScaleFactor(state.ganttScaleFactor); const dayWidth = Math.max(3, Math.round(baseDayWidth * scaleFactor)); return { startDay, endDay, dayWidth, width: Math.max(900, (endDay - startDay + 1) * dayWidth) }; } function ganttTimelineRows() { const milestoneMarkers = state.milestones .filter(item => isDateText(item.date)) .map(item => ({ kind: 'milestone', date: item.date, label: item.name || 'マイルストーン' })); const meetingMarkers = state.meetings.flatMap(item => ( meetingMarkerDates(item).map(date => ({ kind: 'meeting', date, label: item.name || '会議体' })) )); const rows = []; if (milestoneMarkers.length) { rows.push({ label: 'マイルストーン', countLabel: `${milestoneMarkers.length}件`, markers: milestoneMarkers }); } if (meetingMarkers.length) { rows.push({ label: '会議体', countLabel: `${meetingMarkers.length}件`, markers: meetingMarkers }); } return rows; } function ganttTimelineMarkerDates() { return ganttTimelineRows() .flatMap(row => row.markers.map(marker => dateToDay(marker.date))) .filter(day => Number.isFinite(day)); } function meetingMarkerDates(meeting) { const text = [meeting.name, meeting.schedule, meeting.note].map(value => value || '').join(' '); const seen = new Set(); return (text.match(/\d{4}-\d{2}-\d{2}/g) || []).filter(date => { if (!isDateText(date) || seen.has(date)) return false; seen.add(date); return true; }); } function isDateText(value) { const text = String(value || ''); if (!/^\d{4}-\d{2}-\d{2}$/.test(text)) return false; const day = dateToDay(text); return Number.isFinite(day) && dayToDate(day) === text; } function addMonthDay(day) { const date = new Date(day * 86400000); const next = new Date(Date.UTC(date.getUTCFullYear(), date.getUTCMonth() + 1, 1)); return Math.floor(next.getTime() / 86400000); }
